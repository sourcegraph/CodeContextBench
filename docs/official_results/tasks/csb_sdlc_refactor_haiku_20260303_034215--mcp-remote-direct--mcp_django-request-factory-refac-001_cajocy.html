import re
import sys
import os

def replace_in_file(filepath):
    with open(filepath, 'r') as f:
        content = f.read()

    # This handles cases like:
    # from django.test import (
    #     RequestFactory,
    #     ...
    # )
    content = re.sub(

        'from django.test import (',

    content = content.replace(

    )

    content = re.sub(
        r'from django\.test import (.+?)RequestFactory',
        r'from django.test import \1TestRequestBuilder',
        content
    )
    
    # Replace RequestFactory in multi-line imports (after comma)
    content = content.replace(
        '\n    RequestFactory,',
        '\n    TestRequestBuilder,'
    )
    
    # Replace RequestFactory = something with TestRequestBuilder = something
    # (but not the backward compat alias we added)
    
    with open(filepath, 'w') as f:
        f.write(content)
    
    return content

# Process all test files
