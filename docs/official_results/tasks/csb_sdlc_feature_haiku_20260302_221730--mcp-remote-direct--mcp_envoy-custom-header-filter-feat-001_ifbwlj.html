configurable headers into requests and/or responses based on route metadata.

Config::Config(const ProtoConfig& proto_config, const bool per_route) {
  // Process request rules
  for (const auto& rule : proto_config.request_rules()) {
    request_rules_.push_back(rule);
  }

  // Process response rules
  for (const auto& rule : proto_config.response_rules()) {

  // route level. At the virtual or route level, it makes no sense to have an empty
  // config so we throw an error.

                         "either request or response rules");
