## Background

The `runtime.Object` interface is the foundational interface for all Kubernetes API types. It is defined in the `k8s.io/apimachinery/pkg/runtime` package and requires two methods:

```go
type Object interface {
    GetObjectKind() schema.ObjectKind
    DeepCopyObject() Object
}
```

Every Kubernetes resource type (Pod, Deployment, Service, etc.) implements this interface by embedding `metav1.TypeMeta` and implementing the autogenerated `DeepCopyObject()` method.

## Repositories

Two repositories are available under `/workspace/`:

- `/workspace/api/` — kubernetes/api (standard API types: core/v1, apps/v1, etc.)
- `/workspace/apimachinery/` — kubernetes/apimachinery (runtime machinery, metav1, schema)

## Task

Find **all struct types** that implement the `runtime.Object` interface across both repositories. Focus on:

1. **Concrete resource types** (Pod, Deployment, Service, ConfigMap, etc.)
2. **Types with TypeMeta embedded** — the standard pattern for runtime.Object implementors
3. **Types with DeepCopyObject() methods** — either explicit or autogenerated

For each implementor, record:
- `repo`: either `kubernetes/api` or `kubernetes/apimachinery`
- `file`: path relative to the repository root (e.g., `core/v1/types.go`)
- `struct_name`: the name of the struct type (e.g., `Pod`, `Deployment`)

## Inclusion Criteria

**Include:**
- Core resource types (Pod, Deployment, Service, ConfigMap, Node, Namespace, etc.)
- Versioned API types from k8s.io/api packages (core/v1, apps/v1, networking/v1, batch/v1, etc.)
- Types with `metav1.TypeMeta` and `metav1.ObjectMeta` fields
- Types registered in Kubernetes API schemes

**Exclude:**
- The interface definition itself (`runtime.Object` in `runtime/interfaces.go`)
- Internal-only types not exposed in versioned APIs
- List types (e.g., `PodList`, `DeploymentList`) — focus on single-object types only
- Generated code files (`*_generated.go`, `zz_generated.*`)
- Test helper structs and mock types
- Types in vendor directories or third-party packages

package main

import (
	"fmt"
	"os"
	"path/filepath"
	"regexp"
	"strings"
)

func main() {
	var results []map[string]string

	// Search in kubernetes/api
	apiRoot := "/workspace/api"
	err := filepath.Walk(apiRoot, func(path string, info os.FileInfo, err error) error {
		if err != nil {
			return nil
		}
		
		if !strings.HasSuffix(path, "types.go") || strings.Contains(path, "generated") {
			return nil
		}
		
		content, err := os.ReadFile(path)
		if err != nil {
			return nil
		}
		
		text := string(content)
		
		// Find struct definitions with metav1.TypeMeta
		// Pattern: type <Name> struct { ... metav1.TypeMeta ...
		typePattern := regexp.MustCompile(`(?s)type\s+(\w+)\s+struct\s*\{[^}]*metav1\.TypeMeta`)
		matches := typePattern.FindAllStringSubmatch(text, -1)
		
		for _, match := range matches {
			structName := match[1]
			// Skip list types
			if strings.HasSuffix(structName, "List") {
				continue
			}
			relPath := strings.TrimPrefix(path, apiRoot+"/")
			results = append(results, map[string]string{
				"repo":        "kubernetes/api",
				"file":        relPath,
				"struct_name": structName,
			})
		}
		
		return nil
	})
	
	if err != nil {
		fmt.Fprintf(os.Stderr, "Error: %v\n", err)
		os.Exit(1)
	}

	// Search in kubernetes/apimachinery
	amRoot := "/workspace/apimachinery"
	err = filepath.Walk(amRoot, func(path string, info os.FileInfo, err error) error {
		if err != nil {
			return nil
		}
		
		if !strings.HasSuffix(path, ".go") || strings.Contains(path, "generated") || strings.Contains(path, "_test.go") {
			return nil
		}
		
		// Only look in pkg/apis/meta for metav1 types
		if !strings.Contains(path, "pkg/apis/meta") {
			return nil
		}
		
		content, err := os.ReadFile(path)
		if err != nil {
			return nil
		}
		
		text := string(content)
		
		// Find struct definitions with metav1.TypeMeta
		typePattern := regexp.MustCompile(`(?s)type\s+(\w+)\s+struct\s*\{[^}]*TypeMeta`)
		matches := typePattern.FindAllStringSubmatch(text, -1)
		
		for _, match := range matches {
			structName := match[1]
			// Skip list types
			if strings.HasSuffix(structName, "List") {
				continue
			}
			relPath := strings.TrimPrefix(path, amRoot+"/")
			results = append(results, map[string]string{
				"repo":        "kubernetes/apimachinery",
				"file":        relPath,
				"struct_name": structName,
			})
		}
		
		return nil
	})
	
	if err != nil {
		fmt.Fprintf(os.Stderr, "Error: %v\n", err)
		os.Exit(1)
	}

	// Output as JSON
	fmt.Println("[")
	for i, r := range results {
		fmt.Printf("  {\n")
		fmt.Printf("    \"repo\": \"%s\",\n", r["repo"])
		fmt.Printf("    \"file\": \"%s\",\n", r["file"])
		fmt.Printf("    \"struct_name\": \"%s\"\n", r["struct_name"])
		fmt.Printf("  }")
		if i < len(results)-1 {
			fmt.Printf(",")
		}
		fmt.Printf("\n")
	}
	fmt.Println("]")
}
EOF

results = []

# Search in kubernetes/api
api_root = Path("/workspace/api")
for types_file in api_root.rglob("types.go"):
    if "generated" in str(types_file):
        continue
    
    try:
        content = types_file.read_text(encoding='utf-8', errors='ignore')
    except:
        continue
    
    # Find struct definitions with metav1.TypeMeta
    pattern = r'type\s+(\w+)\s+struct\s*\{[^}]*metav1\.TypeMeta'
    matches = re.finditer(pattern, content, re.DOTALL)
    
    for match in matches:
        struct_name = match.group(1)
        # Skip list types
        if struct_name.endswith("List"):
            continue
        
        rel_path = str(types_file.relative_to(api_root))
        results.append({
            "repo": "kubernetes/api",
            "file": rel_path,
            "struct_name": struct_name
        })

# Search in kubernetes/apimachinery
am_root = Path("/workspace/apimachinery")
for py_file in am_root.rglob("*.go"):
    if "generated" in str(py_file) or "_test.go" in str(py_file):
        continue
    
    # Only look in pkg/apis/meta for metav1 types
    if "pkg/apis/meta" not in str(py_file):
        continue
    
    try:
        content = py_file.read_text(encoding='utf-8', errors='ignore')
    except:
        continue
    
    # Find struct definitions with TypeMeta
    pattern = r'type\s+(\w+)\s+struct\s*\{[^}]*TypeMeta'
    matches = re.finditer(pattern, content, re.DOTALL)
    
    for match in matches:
        struct_name = match.group(1)
        # Skip list types
        if struct_name.endswith("List"):
            continue
        
        rel_path = str(py_file.relative_to(am_root))
        results.append({
            "repo": "kubernetes/apimachinery",
            "file": rel_path,
            "struct_name": struct_name
        })

# Sort results
results.sort(key=lambda x: (x["repo"], x["file"], x["struct_name"]))

# Output as JSON
print(json.dumps(results, indent=2))

import json

with open('/workspace/implementors.json') as f:
    data = json.load(f)

# Verify some well-known resource types are included
well_known = ['Pod', 'Deployment', 'Service', 'ConfigMap', 'Node', 'Namespace']

print("Checking for well-known resource types:")
for type_name in well_known:
    matches = [x for x in data if x['struct_name'] == type_name]
    if matches:
        print(f"✓ {type_name}: {len(matches)} version(s)")
        for match in matches:
            print(f"    - {match['file']}")
    else:
        print(f"✗ {type_name}: NOT FOUND")

print(f"\nFull list statistics:")
print(f"Total: {len(data)}")
print(f"Files represented: {len(set(x['file'] for x in data))}")
print(f"Unique struct names: {len(set(x['struct_name'] for x in data))}")

# Most common struct names (versioned types)
from collections import Counter
struct_names = Counter(x['struct_name'] for x in data)
print(f"\nMost frequently occurring struct names (versioned APIs):")
for name, count in struct_names.most_common(10):
    print(f"  {name}: {count} versions")
