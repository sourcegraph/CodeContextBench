// mockTransformer is a test implementation of the Transformer interface
// that can be configured to simulate various encryption/decryption behaviors
type mockTransformer struct {
	encryptFunc func(ctx context.Context, data []byte) ([]byte, error)
	decryptFunc func(ctx context.Context, data []byte) ([]byte, bool, error)
}

func (m *mockTransformer) TransformToStorage(ctx context.Context, data []byte, dataCtx Context) ([]byte, error) {
	if m.encryptFunc != nil {
		return m.encryptFunc(ctx, data)
	}
	// Default: simple XOR transformation for testing
	result := make([]byte, len(data))
	for i, b := range data {
		result[i] = b ^ 0xFF
	}
	return result, nil
}

	if m.decryptFunc != nil {
		return m.decryptFunc(ctx, data)
	}
	// Default: reverse the XOR transformation
	result := make([]byte, len(data))
	for i, b := range data {
		result[i] = b ^ 0xFF
	}
	return result, false, nil
}
