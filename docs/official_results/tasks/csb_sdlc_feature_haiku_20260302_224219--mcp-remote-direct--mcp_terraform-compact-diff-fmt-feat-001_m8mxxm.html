// terraform plan, showing only resource-level changes without attribute-level detail.
type CompactDiffFormatter struct{}

// Format returns a compact string representation of a plan's changes,
// grouped by action type and showing only resource-level information.
func (f *CompactDiffFormatter) Format(plan *plans.Plan, schemas *schemarepo.Schemas) string {
	if plan == nil || plan.Changes == nil {
		return ""
	}

	// Group resources by action
	resources := make(map[plans.Action][]*plans.ResourceInstanceChangeSrc)
	for _, rc := range plan.Changes.Resources {
		action := rc.Action
		resources[action] = append(resources[action], rc)
	}

	// Sort each action's resources
	for action := range resources {
		sort.Slice(resources[action], func(i, j int) bool {
