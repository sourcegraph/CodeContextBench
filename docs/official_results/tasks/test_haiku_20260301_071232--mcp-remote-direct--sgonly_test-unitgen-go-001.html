// mockTransformer is a test transformer that applies simple transformations
type mockTransformer struct {
	encryptPrefix []byte
	shouldFail    bool
	failOnFrom    bool
	failOnTo      bool
}

func (m *mockTransformer) TransformFromStorage(ctx context.Context, data []byte, dataCtx Context) ([]byte, bool, error) {
	if m.failOnFrom || m.shouldFail {
		return nil, false, fmt.Errorf("mock decrypt error")
	}
	// Remove the prefix to get original data
	if bytes.HasPrefix(data, m.encryptPrefix) {
		return bytes.TrimPrefix(data, m.encryptPrefix), false, nil
	}
	return nil, false, fmt.Errorf("data does not have expected prefix")
}

func (m *mockTransformer) TransformToStorage(ctx context.Context, data []byte, dataCtx Context) ([]byte, error) {
	if m.failOnTo || m.shouldFail {
		return nil, fmt.Errorf("mock encrypt error")
	}
	// Add the prefix to simulate encryption
	result := make([]byte, len(m.encryptPrefix)+len(data))
	copy(result, m.encryptPrefix)
	copy(result[len(m.encryptPrefix):], data)
	return result, nil
}
