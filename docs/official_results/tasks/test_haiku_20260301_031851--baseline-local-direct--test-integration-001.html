	client := setupEvaluationTestServer(t)
	defer client.Close()

	body := map[string]interface{}{
		"request_id":    "test-request-1",
		"namespace_key": "production",
		"flag_key":      "test_flag",
		"entity_id":     "user123",
		"context": map[string]string{
			"plan": "pro",
		},
	}

	bodyBytes, err := json.Marshal(body)
	require.NoError(t, err)

	resp, err := http.Post(
		client.URL+"/evaluate/v1/boolean",
		"application/json",
		bytes.NewReader(bodyBytes),
	)
	require.NoError(t, err)
	defer resp.Body.Close()

	// Verify the response has correct status code
	if resp.StatusCode != http.StatusOK {
		bodyContent, _ := io.ReadAll(resp.Body)
		t.Fatalf("expected status 200, got %d: %s", resp.StatusCode, string(bodyContent))
	}

	var result rpcevaluation.BooleanEvaluationResponse
	err = json.NewDecoder(resp.Body).Decode(&result)
	require.NoError(t, err)

	assert.Equal(t, "test-request-1", result.RequestId)
	assert.NotEmpty(t, result.Reason)
}
