1. WRITE /workspace/internal/server/evaluation/duration_tracker.go
package evaluation

import (
	"sort"
	"sync"
	"time"
)

// DurationTracker records evaluation durations and provides statistics.
type DurationTracker struct {
	mu        sync.RWMutex
	durations []time.Duration
}

// Record adds a duration to the tracker.
func (dt *DurationTracker) Record(d time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, d)
}

// Count returns the total number of recorded durations.
func (dt *DurationTracker) Count() int {
	dt.mu.RLock()
	defer dt.mu.RUnlock()
	return len(dt.durations)
}

// Average returns the average duration, or 0 if no durations have been recorded.
func (dt *DurationTracker) Average() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	var total time.Duration
	for _, d := range dt.durations {
		total += d
	}

	return total / time.Duration(len(dt.durations))
}

// P99 returns the 99th percentile duration, or 0 if no durations have been recorded.
func (dt *DurationTracker) P99() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	// Create a copy of durations to sort
	durationsCopy := make([]time.Duration, len(dt.durations))
	copy(durationsCopy, dt.durations)

	// Sort the durations
	sort.Slice(durationsCopy, func(i, j int) bool {
		return durationsCopy[i] < durationsCopy[j]
	})

	// Calculate the 99th percentile index
	// For P99, we want the value at position (count * 0.99)
	// Using ceiling to round up and ensure we get a valid index
	index := (len(durationsCopy)*99 + 99) / 100 - 1
	if index < 0 {
		index = 0
	}
	if index >= len(durationsCopy) {
		index = len(durationsCopy) - 1
	}

	return durationsCopy[index]
}
