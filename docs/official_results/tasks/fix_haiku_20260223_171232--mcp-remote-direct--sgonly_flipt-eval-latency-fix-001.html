// DurationTracker records the wall-clock duration of evaluations and provides
// aggregate statistics including count, average, and P99 latency.
type DurationTracker struct {
	mu        sync.RWMutex
	durations []time.Duration
}

// NewDurationTracker creates a new duration tracker.
func NewDurationTracker() *DurationTracker {
	return &DurationTracker{
		durations: make([]time.Duration, 0),
	}
}

// Record records a duration from an evaluation.
func (dt *DurationTracker) Record(duration time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, duration)
}

// Stats returns the count, average, and P99 latency of all recorded evaluations.
func (dt *DurationTracker) Stats() (count int, avg time.Duration, p99 time.Duration) {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0, 0, 0
	}

	count = len(dt.durations)

	// Calculate average
	var total time.Duration
	for _, d := range dt.durations {
		total += d
	}
	avg = total / time.Duration(count)

	// Calculate P99
	sorted := make([]time.Duration, len(dt.durations))
	copy(sorted, dt.durations)
	sort.Slice(sorted, func(i, j int) bool {
		return sorted[i] < sorted[j]
	})

	p99Index := (count * 99) / 100
	if p99Index >= count {
		p99Index = count - 1
	}
	p99 = sorted[p99Index]

	return count, avg, p99
}
