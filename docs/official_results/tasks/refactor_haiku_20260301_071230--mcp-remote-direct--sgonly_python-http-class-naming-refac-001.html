            new_content = re.sub(

                f': {new_name}',
                new_content
            )
            new_content = re.sub(
                rf'->\s*{old_name}\b',
                f'-> {new_name}',
                new_content
            )
            
            # Replace in isinstance checks
            new_content = re.sub(
                rf'isinstance\(([^,]+),\s*{old_name}\b',
                rf'isinstance(\1, {new_name}',
                new_content
            )
            
            # Replace in docstrings and comments with :class: or ~
            new_content = re.sub(
                rf':class:`~?{old_name}`',
                f':class:`~{new_name}`',
                new_content
            )
            
            # Replace in other docstring/documentation references
            new_content = re.sub(
                rf':class:`{old_name}`',
                f':class:`{new_name}`',
                new_content
            )
            
            # Replace parenthesized references like (HttpRequest)
            new_content = re.sub(
                rf'\({old_name}\)',
                f'({new_name})',
                new_content
            )
            
            # Replace in extends/subclass declarations
            new_content = re.sub(
                rf'class\s+(\w+)\({old_name}\)',
                rf'class \1({new_name})',
                new_content
            )
            
            if new_content != content:
                with open(filepath, 'w', encoding='utf-8') as f:
                    f.write(new_content)
                changes.append(str(filepath))
                print(f"Modified: {filepath}")
    
    return changes
