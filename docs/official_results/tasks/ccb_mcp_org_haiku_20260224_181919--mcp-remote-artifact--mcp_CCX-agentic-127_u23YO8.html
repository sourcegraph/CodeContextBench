/* Main routine to count basic blocks per function.  */

static unsigned int
count_basic_blocks_per_function ()
{
  basic_block bb;
  int bb_count = 0;

  FOR_EACH_BB_FN (bb, cfun)
    {
      bb_count++;
    }

  if (dump_file && (dump_flags & TDF_DETAILS))
    {
      fprintf (dump_file, "Function %s has %d basic blocks\n",
	       function_name (cfun), bb_count);
    }

  return 0;
}
