14. $ cat > /tmp/analyze_all.sh << 'EOF'
#!/bin/bash

# Function to check patterns in a file
check_patterns() {
    local file=$1
    local dir=$2
    local patterns=""
    
    if grep -q "INITIALIZE_PASS_BEGIN" "$file"; then
        patterns="$patterns INITIALIZE_PASS_BEGIN/END"
    elif grep -q "INITIALIZE_PASS" "$file"; then
        patterns="$patterns INITIALIZE_PASS"
    fi
    
    if grep -q "public FunctionPass" "$file"; then
        patterns="$patterns FunctionPass"
    fi
    
    if grep -q "public ModulePass" "$file"; then
        patterns="$patterns ModulePass"
    fi
    
    if grep -q "FunctionPass \*" "$file"; then
        patterns="$patterns create*Pass(FunctionPass*)"
    fi
    
    if grep -q "ModulePass \*" "$file"; then
        patterns="$patterns create*Pass(ModulePass*)"
    fi
    
    echo "$(basename "$file")|$dir|$patterns"
}

echo "=== COMPREHENSIVE ANALYSIS ==="
for file in /workspace/llvm-project--a8f3c97d/llvm/lib/Transforms/Scalar/*.cpp; do
    check_patterns "$file" "Scalar"
done | grep -v "^|"

for file in /workspace/llvm-project--a8f3c97d/llvm/lib/Transforms/IPO/*.cpp; do
    check_patterns "$file" "IPO"
done | grep -v "^|"
EOF
