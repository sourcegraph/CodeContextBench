// CompactDiffFormatter produces a condensed, summary-style diff output for
// terraform plan, showing only resource-level changes without attribute-level
// detail.
type CompactDiffFormatter struct{}

// Format returns a compact diff summary of the given plan. It outputs one line
// per resource change in the format "[action] resource_type.name", groups
// changes by action type, and shows summary counts at the end.
func (f *CompactDiffFormatter) Format(plan *plans.Plan, schemas *schemarepo.Schemas) string {
	if plan == nil || plan.Changes == nil {
		return ""
	}

	var buf bytes.Buffer

	// Group resources by action
	changesByAction := make(map[plans.Action][]*plans.ResourceInstanceChangeSrc)
	for _, rc := range plan.Changes.Resources {
		changesByAction[rc.Action] = append(changesByAction[rc.Action], rc)
	}

	// Define the order of actions for output
	actionOrder := []plans.Action{
		plans.Create,
		plans.Update,
		plans.Delete,
		plans.DeleteThenCreate,
		plans.CreateThenDelete,
		plans.Read,
		plans.Forget,
		plans.CreateThenForget,
		plans.NoOp,
	}

	counts := make(map[plans.Action]int)
