import re

with open('/app/pyproject.toml', 'r') as f:
    content = f.read()

# Check for common TOML errors
errors = []

# Check if all brackets are balanced
open_brackets = content.count('[')
close_brackets = content.count(']')
if open_brackets != close_brackets:
    errors.append(f"Unbalanced brackets: {open_brackets} open, {close_brackets} close")

# Check for required sections
if '[tool.poetry]' not in content:
    errors.append("Missing [tool.poetry] section")
if '[tool.poetry.dependencies]' not in content:
    errors.append("Missing [tool.poetry.dependencies] section")
if '[tool.poetry.group.dev.dependencies]' not in content:
    errors.append("Missing [tool.poetry.group.dev.dependencies] section")
if '[build-system]' not in content:
    errors.append("Missing [build-system] section")

if errors:
    print("✗ Errors found:")
    for error in errors:
        print(f"  - {error}")
else:
    print("✓ pyproject.toml structure looks valid")
    
    # Extract and list dependencies
    deps_section = re.search(r'\[tool\.poetry\.dependencies\](.*?)\[', content, re.DOTALL)
    if deps_section:
        deps = deps_section.group(1).strip().split('\n')
        prod_deps = [d.strip() for d in deps if d.strip() and not d.strip().startswith('#')]
        print(f"\n✓ Found {len(prod_deps)} production dependencies")
        
    dev_deps_section = re.search(r'\[tool\.poetry\.group\.dev\.dependencies\](.*?)\[', content, re.DOTALL)
    if dev_deps_section:
        dev_deps = dev_deps_section.group(1).strip().split('\n')
        dev_deps_clean = [d.strip() for d in dev_deps if d.strip() and not d.strip().startswith('#')]
        print(f"✓ Found {len(dev_deps_clean)} development dependencies")
