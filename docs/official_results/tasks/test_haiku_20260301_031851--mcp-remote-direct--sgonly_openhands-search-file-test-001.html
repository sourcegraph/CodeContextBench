    """Capture stdout output from a function call.

    Args:
        operation: A callable that produces output to stdout

    Returns:
        str: The captured output
    """
    with io.StringIO() as buf:
        with contextlib.redirect_stdout(buf):
            operation()
        return buf.getvalue()
