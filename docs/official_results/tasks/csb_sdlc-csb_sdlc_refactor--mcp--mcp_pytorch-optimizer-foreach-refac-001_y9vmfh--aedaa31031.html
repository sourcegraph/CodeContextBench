    state_steps: list[Tensor],
    step_value: float = 1.0,
) -> None:
    """Update state steps, with special handling for CPU tensors.

        state_steps: List of state step tensors to update.

        )

        # For non-CPU tensors, pass the scalar directly
        torch._foreach_add_(state_steps, step_value)
