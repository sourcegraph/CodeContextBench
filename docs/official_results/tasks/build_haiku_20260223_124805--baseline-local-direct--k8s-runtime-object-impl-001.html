## Background

The `runtime.Object` interface is the foundational interface for all Kubernetes API types. It is defined in the `k8s.io/apimachinery/pkg/runtime` package and requires two methods:

```go
type Object interface {
    GetObjectKind() schema.ObjectKind
    DeepCopyObject() Object
}
```

Every Kubernetes resource type (Pod, Deployment, Service, etc.) implements this interface by embedding `metav1.TypeMeta` and implementing the autogenerated `DeepCopyObject()` method.

- `/workspace/api/` — kubernetes/api (standard API types: core/v1, apps/v1, etc.)
- `/workspace/apimachinery/` — kubernetes/apimachinery (runtime machinery, metav1, schema)

func findImplementorsInDir(rootDir, repo string) []Implementor {
	var implementors []Implementor
	typePattern := regexp.MustCompile(`^type\s+(\w+)\s+struct\s*\{`)
	hasTypeMetaPattern := regexp.MustCompile(`metav1\.TypeMeta`)
	isListType := regexp.MustCompile(`List$`)

	filepath.Walk(rootDir, func(path string, info os.FileInfo, err error) error {
		if err != nil || info.IsDir() {
			return err
		}

		// Only process types.go files, exclude generated files
		if !strings.HasSuffix(path, "types.go") {
			return nil
		}
		if strings.Contains(path, "generated") || strings.Contains(path, "test") {
			return nil
		}

		file, err := os.Open(path)
		if err != nil {
			return nil
		}
		defer file.Close()

		scanner := bufio.NewScanner(file)
		inStruct := false
		currentType := ""
		structContent := ""
		lineNum := 0

		for scanner.Scan() {
			line := scanner.Text()
			lineNum++

			// Check for type declaration
			if match := typePattern.FindStringSubmatch(line); match != nil && !inStruct {
				currentType = match[1]
				inStruct = true
				structContent = line
				continue
			}

			if inStruct {
				structContent += "\n" + line

				// Look for closing brace
				if line == "}" {
					// Check if this struct has TypeMeta and is not a List type
					if hasTypeMetaPattern.MatchString(structContent) && !isListType.MatchString(currentType) {
						relPath, _ := filepath.Rel(rootDir, path)
						implementors = append(implementors, Implementor{

							StructName: currentType,
						})
					}
					inStruct = false
					currentType = ""
					structContent = ""
				}
			}
		}

		return nil
	})

	return implementors
}
