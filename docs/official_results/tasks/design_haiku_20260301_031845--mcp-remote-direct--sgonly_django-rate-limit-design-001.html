"""
Rate limiting middleware to restrict HTTP requests by IP address.

Tracks requests per client IP and returns a 403 Forbidden response when
the rate limit is exceeded within a configured time window.
"""

import time

from django.conf import settings
from django.http import HttpResponseForbidden
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Rate limit incoming HTTP requests by IP address.

    Tracks the number of requests from each IP address within a time window
    and returns a 403 Forbidden response when the rate limit is exceeded.

    Configuration via Django settings:
    - RATE_LIMIT_REQUESTS: Maximum requests per window (default: 100)
    - RATE_LIMIT_WINDOW: Time window in seconds (default: 3600)
    """

    def __init__(self, get_response):
        super().__init__(get_response)
        self.rate_limit_requests = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        self.rate_limit_window = getattr(settings, "RATE_LIMIT_WINDOW", 3600)
        self.request_times = {}
        self.request_count = 0

    def get_client_ip(self, request):
        """
        Extract the client's IP address from the request.
        """
        x_forwarded_for = request.META.get("HTTP_X_FORWARDED_FOR")
        if x_forwarded_for:
            ip = x_forwarded_for.split(",")[0]
        else:
