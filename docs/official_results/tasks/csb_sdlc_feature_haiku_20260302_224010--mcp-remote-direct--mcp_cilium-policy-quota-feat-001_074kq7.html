// Cell provides the PolicyQuotaController as a hive component.
var Cell = cell.Module(
	"policy-quota-controller",
	"Enforces per-namespace limits on CiliumNetworkPolicy resources",
	cell.Invoke(registerPolicyQuotaController),
)

type PolicyQuotaControllerParams struct {
	cell.In

	Lifecycle cell.Lifecycle
	Config    *option.DaemonConfig
	Logger    logrus.FieldLogger

	CiliumNetworkPolicies resource.Resource[*cilium_v2.CiliumNetworkPolicy]
}

func registerPolicyQuotaController(params PolicyQuotaControllerParams) {
	if params.Config == nil {
		return
	}

	controller := NewPolicyQuotaController(
		10, // Default max policies per namespace, can be made configurable
		params.CiliumNetworkPolicies,
	)

	params.Lifecycle.Append(cell.Hook{
		OnStart: func(startCtx cell.HookContext) error {
			go controller.Run(startCtx)
			return nil
		},
	})
}
