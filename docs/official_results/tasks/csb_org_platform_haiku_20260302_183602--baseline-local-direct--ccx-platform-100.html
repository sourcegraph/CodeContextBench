#!/bin/bash

# Function to extract Deprecated symbols from a file
extract_symbols() {
    local file=$1
    local dir=$2

    # This includes const definitions and struct fields
    grep -n "Deprecated[A-Za-z0-9_]*" "$file" | grep -v "^\s*//" | grep -v "zz_generated" | while read line; do

        if echo "$content" | grep -q "^\s*Deprecated[A-Za-z0-9_]*\s*=\|^\s*Deprecated[A-Za-z0-9_]*\s\+[A-Za-z]"; then
            symbol=$(echo "$content" | sed -n 's/^\s*\(Deprecated[A-Za-z0-9_]*\).*/\1/p')
            if [ ! -z "$symbol" ]; then
                echo "$dir|$file|$linenum|$symbol"
            fi
        fi
    done
}
