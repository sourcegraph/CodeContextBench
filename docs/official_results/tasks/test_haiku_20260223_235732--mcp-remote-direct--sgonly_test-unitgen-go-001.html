// mockTransformer is a simple transformer implementation for testing
type mockTransformer struct {
	transformFromFunc func(ctx context.Context, data []byte, dataCtx Context) (out []byte, stale bool, err error)
	transformToFunc   func(ctx context.Context, data []byte, dataCtx Context) (out []byte, err error)
}

func (m *mockTransformer) TransformFromStorage(ctx context.Context, data []byte, dataCtx Context) (out []byte, stale bool, err error) {
	if m.transformFromFunc != nil {
		return m.transformFromFunc(ctx, data, dataCtx)
	}
	return data, false, nil
}

func (m *mockTransformer) TransformToStorage(ctx context.Context, data []byte, dataCtx Context) (out []byte, err error) {
	if m.transformToFunc != nil {
		return m.transformToFunc(ctx, data, dataCtx)
	}
	return data, nil
}

// identityTransformer returns the input unchanged
func identityTransformer() Transformer {
	return &mockTransformer{
		transformFromFunc: func(ctx context.Context, data []byte, dataCtx Context) ([]byte, bool, error) {
			return data, false, nil
		},
		transformToFunc: func(ctx context.Context, data []byte, dataCtx Context) ([]byte, error) {
			return data, nil
		},
	}
}
