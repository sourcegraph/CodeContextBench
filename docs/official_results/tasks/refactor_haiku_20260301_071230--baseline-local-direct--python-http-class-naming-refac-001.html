import os
import re
from pathlib import Path

flask_dir = Path("/ccb_crossrepo/src/flask")

def rename_flask_request():
    """Rename Flask's Request class to HTTPRequest."""
    
    # First, rename the class definition in wrappers.py
    wrappers_file = flask_dir / "src/flask/wrappers.py"
    content = wrappers_file.read_text()
    content = content.replace("class Request(RequestBase):", "class HTTPRequest(RequestBase):")
    wrappers_file.write_text(content)
    print(f"Updated {wrappers_file}: class definition")
    
    # Update all Python files
    for py_file in flask_dir.rglob("*.py"):
        if "/.git/" in str(py_file):
            continue
        original = py_file.read_text()
        modified = original
        
        # Replace imports
        modified = re.sub(r'from \.wrappers import Request\b', 'from .wrappers import HTTPRequest', modified)
        modified = re.sub(r'from flask\.wrappers import Request\b', 'from flask.wrappers import HTTPRequest', modified)
        modified = re.sub(r'from \.wrappers import Request as Request', 'from .wrappers import HTTPRequest as HTTPRequest', modified)
        modified = re.sub(r'from flask import Request\b', 'from flask import HTTPRequest', modified)
        
        # Replace in type annotations and references
        # Be careful to only replace Flask's Request, not other uses
        # Replace ": Request" with ": HTTPRequest" (type annotation)
        modified = re.sub(r':\s+Request\b(?!\()', ': HTTPRequest', modified)
        # Replace "Request[" with "HTTPRequest["
        modified = re.sub(r'\bRequest\[', 'HTTPRequest[', modified)
        # Replace "Request |" with "HTTPRequest |"
        modified = re.sub(r'\bRequest\s+\|', 'HTTPRequest |', modified)
        # Replace "| Request" with "| HTTPRequest"
        modified = re.sub(r'\|\s+Request\b', '| HTTPRequest', modified)
        # Replace ", Request)" with ", HTTPRequest)"
        modified = re.sub(r',\s+Request\)', ', HTTPRequest)', modified)
        # Replace "(Request)" with "(HTTPRequest)"
        modified = re.sub(r'\(\s*Request\s*\)', '(HTTPRequest)', modified)
        # Replace "RequestProxy[Request]" with "RequestProxy[HTTPRequest]"
        modified = re.sub(r'RequestProxy\[Request\]', 'RequestProxy[HTTPRequest]', modified)
        
        if modified != original:
