// Silence represents a silence in the Alertmanager.
type Silence struct {
	ID        string            `json:"id,omitempty"`
	Matchers  []SilenceMatcher  `json:"matchers"`
	StartsAt  time.Time         `json:"startsAt"`
	EndsAt    time.Time         `json:"endsAt"`
	CreatedBy string            `json:"createdBy"`
	Comment   string            `json:"comment"`
}

// SilenceMatcher matches labels for silencing alerts.
type SilenceMatcher struct {
	Name    string `json:"name"`
	Value   string `json:"value"`
	IsRegex bool   `json:"isRegex"`
}

// BulkSilenceRequest represents a request to create multiple silences.
type BulkSilenceRequest struct {
	Silences []Silence `json:"silences"`
}

// BulkSilenceResponse represents the response from creating multiple silences.
type BulkSilenceResponse struct {
	SilenceIDs []string `json:"silenceIds"`
}

// BulkSilenceError represents an error response with detailed failure information.
type BulkSilenceError struct {
	Index   int    `json:"index"`
	Message string `json:"message"`
}

// validateSilence validates a silence object.
func validateSilence(s *Silence) error {
	if s == nil {
		return errors.New("silence cannot be nil")
	}
	if len(s.Matchers) == 0 {
		return errors.New("silence must have at least one matcher")
	}
	for i, m := range s.Matchers {
		if m.Name == "" {
			return fmt.Errorf("matcher %d: name cannot be empty", i)
		}
	}
	if s.StartsAt.IsZero() {
		return errors.New("startsAt cannot be empty")
	}
	if s.EndsAt.IsZero() {
		return errors.New("endsAt cannot be empty")
	}
	if s.EndsAt.Before(s.StartsAt) {
		return errors.New("endsAt must be after startsAt")
	}
	return nil
}
