Lookup sandbox in store\n    sandbox, err := c.sandboxStore.Get(podSandboxID)\n\n    // 2. Stop all containers in sandbox (forced termination)\n    for _, container := range c.containerStore.ListByPodID(podSandboxID) {\n        c.StopContainer(ctx, container.ID, gracePeriod)\n    }\n\n    // 3. Stop sandbox task in containerd\n    // - Send SIGTERM to sandbox\n    // - Wait for grace period\n    // - Force kill if still running\n\n    // 4.
