# Align Solr Boolean Clause Limit with Reading-Log Filter Cap

**Repository:** internetarchive/openlibrary
**Language:** Python
**Difficulty:** hard

## Problem

The reading-log search can generate very large boolean queries. The application enforces a maximum cap defined in `openlibrary/core/bookshelves.py`, while Solr enforces its own maximum via the `-Dsolr.max.booleanClauses=30000` option inside the `SOLR_OPTS` environment variable in `docker-compose.yml`. If these values drift out of sync, searches may fail or behave inconsistently. When Solr's configured limit is lower than the backend cap, large reading-log searches trigger errors or return incomplete results.

## Key Components

- `openlibrary/core/bookshelves.py` — backend cap for reading-log boolean clause count
- `docker-compose.yml` — Solr service definition with `SOLR_OPTS` environment variable containing `-Dsolr.max.booleanClauses`

## Task

1. Locate the boolean clause cap in `openlibrary/core/bookshelves.py`
2. Locate the Solr boolean clause limit in `docker-compose.yml` under `SOLR_OPTS`
3. Align both values to ensure consistency (both should be 30000)
4. Verify that large reading-log queries can execute without errors
5. Run existing tests to ensure no regressions

## Success Criteria

- `bookshelves.py` and `docker-compose.yml` define the same boolean clause limit
- Large reading-log searches execute without Solr errors
- All existing tests pass

import yaml
import re

def test_solr_boolean_clauses_alignment():
    """Test that all compose files have max.booleanClauses=30000"""
    
    # Define the expected value
    expected_value = 30000
    
    # Check all compose files
    compose_files = [
        '/workspace/compose.yaml',
        '/workspace/compose.override.yaml',
        '/workspace/compose.production.yaml',
        '/workspace/scripts/solr_builder/compose.yaml',
    ]
    
    # Also check compose.production.yaml's solr_prod service
    files_with_services = {
        '/workspace/compose.yaml': 'solr',
        '/workspace/compose.override.yaml': 'solr',
        '/workspace/compose.production.yaml': 'solr',
        '/workspace/scripts/solr_builder/compose.yaml': ['solr', 'solr_prod'],
    }
    
    print("Testing SOLR_OPTS max.booleanClauses alignment:")
    print("=" * 70)
    
    all_passed = True
    
    for file_path, services in files_with_services.items():
        if isinstance(services, str):
            services = [services]
        
        try:
            with open(file_path) as f:
                compose = yaml.safe_load(f)
            
            for service_name in services:
                if service_name not in compose.get('services', {}):
                    print(f"✗ {file_path} - service '{service_name}' not found")
                    all_passed = False
                    continue
                
                service = compose['services'][service_name]
                env_vars = service.get('environment', [])
                
                # Find SOLR_OPTS
                solr_opts = None
                for var in env_vars:
                    if isinstance(var, str) and var.startswith('SOLR_OPTS='):
                        solr_opts = var
                        break
                
                if solr_opts is None:
                    # solr service without SOLR_OPTS - check if it's using defaults
                    print(f"✗ {file_path} ({service_name}) - SOLR_OPTS not found")
                    all_passed = False
                    continue
                
                # Extract max.booleanClauses value
                match = re.search(r'-Dsolr\.max\.booleanClauses=(\d+)', solr_opts)
                
                if match:
                    value = int(match.group(1))
                    if value == expected_value:
                        print(f"✓ {file_path} ({service_name}) = {value}")
                    else:
                        print(f"✗ {file_path} ({service_name}) = {value} (expected {expected_value})")
                        all_passed = False
                else:
                    print(f"✗ {file_path} ({service_name}) - max.booleanClauses not found")
                    all_passed = False
        
        except Exception as e:
            print(f"✗ {file_path} - ERROR: {e}")
            all_passed = False
    
    print("=" * 70)
    
    # Also check bookshelves.py
    print("\nTesting bookshelves.py FILTER_BOOK_LIMIT:")
    print("=" * 70)
    
    with open('/workspace/openlibrary/core/bookshelves.py') as f:
        content = f.read()
    
    match = re.search(r'FILTER_BOOK_LIMIT:\s*Final\s*=\s*(\d+_?\d*)', content)
    if match:
        value_str = match.group(1)
        value = int(value_str.replace('_', ''))
        if value == expected_value:
            print(f"✓ FILTER_BOOK_LIMIT = {value_str} ({value})")
        else:
            print(f"✗ FILTER_BOOK_LIMIT = {value_str} ({value}, expected {expected_value})")
            all_passed = False
    else:
        print("✗ FILTER_BOOK_LIMIT not found in bookshelves.py")
        all_passed = False
    
    print("=" * 70)
    
    if all_passed:
        print("\n✓ SUCCESS: All Solr boolean clause limits are properly aligned at 30,000")
        return 0
    else:
        print("\n✗ FAILURE: Some configurations are not aligned")
        return 1

exit(test_solr_boolean_clauses_alignment())
