	"time"

// DurationTracker records evaluation durations and provides aggregate statistics
// in a thread-safe manner.
type DurationTracker struct {
	mu        sync.RWMutex
	durations []time.Duration
}

// NewDurationTracker creates a new DurationTracker.
func NewDurationTracker() *DurationTracker {
	return &DurationTracker{
		durations: make([]time.Duration, 0),
	}
}

// Record records a duration for an evaluation.
func (dt *DurationTracker) Record(d time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, d)
}

// Count returns the total number of recorded evaluations.
func (dt *DurationTracker) Count() int {
	dt.mu.RLock()
	defer dt.mu.RUnlock()
	return len(dt.durations)
}

// Average returns the average duration across all recorded evaluations.
// Returns 0 if no durations have been recorded.
func (dt *DurationTracker) Average() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	var total time.Duration
	for _, d := range dt.durations {
		total += d
	}

	return total / time.Duration(len(dt.durations))
}

// P99 returns the 99th percentile duration.
// Returns 0 if no durations have been recorded.
func (dt *DurationTracker) P99() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	// Create a sorted copy of durations
	sorted := make([]time.Duration, len(dt.durations))
	copy(sorted, dt.durations)
	sort.Slice(sorted, func(i, j int) bool {
		return sorted[i] < sorted[j]
	})

	// Calculate the index for the 99th percentile
	// Using linear interpolation method
	index := int(float64(len(sorted)-1) * 0.99)
	if index >= len(sorted) {
		index = len(sorted) - 1
	}

	return sorted[index]
}

import (
	"sort"

	"time"

// DurationTracker records evaluation durations and provides aggregate statistics
// in a thread-safe manner.
type DurationTracker struct {
	mu        sync.RWMutex
	durations []time.Duration
}

// NewDurationTracker creates a new DurationTracker.
func NewDurationTracker() *DurationTracker {
	return &DurationTracker{
		durations: make([]time.Duration, 0),
	}
}

// Record records a duration for an evaluation.
func (dt *DurationTracker) Record(d time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, d)
}

// Count returns the total number of recorded evaluations.
func (dt *DurationTracker) Count() int {
	dt.mu.RLock()
	defer dt.mu.RUnlock()
	return len(dt.durations)
}

// Average returns the average duration across all recorded evaluations.
// Returns 0 if no durations have been recorded.
func (dt *DurationTracker) Average() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	var total time.Duration
	for _, d := range dt.durations {
		total += d
	}

	return total / time.Duration(len(dt.durations))
}

// P99 returns the 99th percentile duration.
// Returns 0 if no durations have been recorded.
func (dt *DurationTracker) P99() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	// Create a sorted copy of durations
	sorted := make([]time.Duration, len(dt.durations))
	copy(sorted, dt.durations)
	sort.Slice(sorted, func(i, j int) bool {
		return sorted[i] < sorted[j]
	})

	// Calculate the index for the 99th percentile
	// Using linear interpolation method
	index := int(float64(len(sorted)-1) * 0.99)
	if index >= len(sorted) {
		index = len(sorted) - 1
	}

	return sorted[index]
}
