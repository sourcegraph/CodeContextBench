import re
from pathlib import Path

        content = f.read()
    
    issues = []
    
    # Check for old names (word boundaries to avoid false positives)
    patterns = [

    ]

        matches = list(re.finditer(pattern, content))

            issues.append(f"  {name}: {len(matches)} occurrences")
    
    return issues

def main():
    workspace = Path('/workspace/numpy')

                count = int(issue.split(':')[1].strip().split()[0])
                total_old_names += count
    
    if all_issues:
        print("❌ Found remaining old function names:")

if __name__ == '__main__':
