// terraform plan, showing only resource-level changes without attribute-level detail.
type CompactDiffFormatter struct{}

// Format returns a compact diff summary of the plan, grouped by action type
// and with a summary count at the end.
func (f *CompactDiffFormatter) Format(plan *plans.Plan, schemas *schemarepo.Schemas) string {
	if plan == nil || plan.Changes == nil {
		return ""
	}

	// Group changes by action
	changes := make(map[plans.Action][]*addrs.AbsResourceInstance)

	for _, rc := range plan.Changes.Resources {
		// Skip deposed resources and no-op changes
		if rc.DeposedKey != states.NotDeposed || rc.Action == plans.NoOp {
			continue
		}

		changes[rc.Action] = append(changes[rc.Action], &rc.Addr)
	}

	if len(changes) == 0 {
