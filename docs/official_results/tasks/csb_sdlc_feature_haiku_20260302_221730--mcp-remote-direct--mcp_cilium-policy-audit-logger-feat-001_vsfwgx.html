// PolicyAuditLogger provides structured logging of policy evaluation decisions
// for audit and debugging purposes.
type PolicyAuditLogger struct {
	lock.Mutex
	logger   *logrus.Logger
	logLevel logrus.Level
}

// PolicyDecision represents the result of a policy evaluation
type PolicyDecision int

const (
	PolicyDecisionAllow PolicyDecision = iota
	PolicyDecisionDeny
)

// String returns the string representation of a policy decision
func (pd PolicyDecision) String() string {
	switch pd {
	case PolicyDecisionAllow:
		return "ALLOW"
	case PolicyDecisionDeny:
		return "DENY"
	default:
		return "UNKNOWN"
	}
}

// NewPolicyAuditLogger creates a new PolicyAuditLogger with the specified log level and output.
// If output is nil, logs are written to os.Stderr.
func NewPolicyAuditLogger(level logrus.Level, output io.Writer) *PolicyAuditLogger {
	logger := logrus.New()
	logger.SetLevel(level)

	// Set output
	if output == nil {
		output = os.Stderr
	}
	logger.SetOutput(output)

	// Configure JSON formatting for structured audit logs
	logger.SetFormatter(&logrus.JSONFormatter{})

	return &PolicyAuditLogger{
		logger:   logger,
		logLevel: level,
	}
}
