	creates := []*plans.ResourceInstanceChangeSrc{}
	updates := []*plans.ResourceInstanceChangeSrc{}
	deletes := []*plans.ResourceInstanceChangeSrc{}
	replaces := []*plans.ResourceInstanceChangeSrc{}
	noop := []*plans.ResourceInstanceChangeSrc{}

	for _, change := range plan.Changes.Resources {
		// Skip deposed objects (only format the current object)
		if change.DeposedKey != "" {
			continue
		}

		case plans.Create:
			creates = append(creates, change)
