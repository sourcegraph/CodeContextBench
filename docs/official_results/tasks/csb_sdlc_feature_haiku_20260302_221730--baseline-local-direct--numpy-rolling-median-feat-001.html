import numpy as np

# 1D array, valid mode
a = np.array([1, 2, 3, 4, 5])
result = np.rolling_median(a, 3)

# result = [1.5, 2., 3., 4., 4.5]

# result = [[1.5, 2.5, 3.5],
#           [5.5, 6.5, 7.5]]
