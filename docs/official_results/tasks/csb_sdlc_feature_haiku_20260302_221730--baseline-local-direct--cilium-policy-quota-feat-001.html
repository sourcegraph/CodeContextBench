// Cell provides the PolicyQuotaController as a hive module.
var Cell = cell.Module(
	"policy-quota",
	"Manages per-namespace policy quota enforcement",

	cell.Config(QuotaConfig{
		MaxPoliciesPerNamespace: 0, // 0 = unlimited by default
	}),
	cell.Provide(newPolicyQuotaController),
)

// QuotaConfig contains the configuration for the PolicyQuotaController.
type QuotaConfig struct {
	MaxPoliciesPerNamespace int
}

// QuotaControllerParams contains the dependencies for PolicyQuotaController.
type QuotaControllerParams struct {
	cell.In

	Logger                logrus.FieldLogger
	Config                QuotaConfig
	CiliumNetworkPolicies resource.Resource[*cilium_v2.CiliumNetworkPolicy]
}

// newPolicyQuotaController creates a new PolicyQuotaController instance.
func newPolicyQuotaController(params QuotaControllerParams) *PolicyQuotaController {
	return NewPolicyQuotaController(
		params.Logger,
		params.CiliumNetworkPolicies,
		params.Config.MaxPoliciesPerNamespace,
	)
}
