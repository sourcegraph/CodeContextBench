package evaluation

import (
	"sync"
)

// Metrics holds the evaluation metrics counters.
type Metrics struct {
	Total      int64
	MatchCount int64
	ErrorCount int64
}

// Tracker tracks evaluation metrics in a thread-safe manner.
type Tracker struct {
	mu         sync.Mutex
	total      int64
	matchCount int64
	errorCount int64
}

// NewTracker creates a new Tracker instance.
func NewTracker() *Tracker {
	return &Tracker{}
}

// RecordEvaluation records the outcome of an evaluation.
// If matched is true, increments the match count.
// If err is not nil, increments the error count.
// Always increments the total evaluation count.
func (t *Tracker) RecordEvaluation(matched bool, err error) {
	t.mu.Lock()
	defer t.mu.Unlock()

	t.total++
	if matched {
		t.matchCount++
	}
	if err != nil {
		t.errorCount++
	}
}

// GetMetrics returns a copy of the current metrics.
func (t *Tracker) GetMetrics() Metrics {
	t.mu.Lock()
	defer t.mu.Unlock()

	return Metrics{
		Total:      t.total,
		MatchCount: t.matchCount,
		ErrorCount: t.errorCount,
	}
}
