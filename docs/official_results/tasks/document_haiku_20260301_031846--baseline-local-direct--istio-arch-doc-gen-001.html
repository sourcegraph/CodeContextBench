policies\n\n    // Security and policy\n    AuthnPolicies *AuthenticationPolicies      // RequestAuthentication, PeerAuthentication\n    AuthzPolicies *AuthorizationPolicies       // AuthorizationPolicy\n\n    // Extension points\n    EnvoyFilters envoyFiltersByNamespace       // Patches to Envoy config\n    WasmPlugins wasmPluginsByNamespace         // WebAssembly module injection\n    Telemetry *Telemetries                     // Telemetry policies\n\n    // Mesh-wide configuration\n    Mesh *meshconfig.MeshConfig                // Global mesh configuration\n    Networks *meshconfig.MeshNetworks          // Multi-cluster network topology\n\n    // Caching and state\n    JwtKeyResolver *JwksResolver               // JWKS key resolver\n    NetworkManager *NetworkManager             // Network topology information\n\n    // Metrics\n    ProxyStatus map[string]map[string]ProxyPushStatus  // Push results by error code\n}\n```\n\n#### Service Index\n\nThe `serviceIndex` enables efficient service lookups:\n\n```go\ntype serviceIndex struct {\n    // Services visible across namespaces (exportTo: \"*\")\n    public []*Service\n\n    // Namespace-specific services (exportTo: \".\")\n    privateByNamespace map[string][]*Service\n\n    // Services explicitly exported to specific namespaces\n    exportedToNamespace map[string][]*Service\n\n    // Fast lookup: hostname \u2192 namespace \u2192 service\n    HostnameAndNamespace map[host.Name]map[string]*Service\n\n    // Pre-cached endpoints grouped by service key and port\n    instancesByPort map[string]map[int][]*IstioEndpoint\n}\n```\n\n**Why indexing matters**: During RDS (route) generation, a proxy may request all services; the index allows O(1) lookups rather than O(n) scans.\n\n#### Virtual Service Index\n\n```go\ntype virtualServiceIndex struct {\n    // VS public to all namespaces, indexed by gateway\n    publicByGateway map[string][]config.Config\n\n    // VS scoped to namespaces, indexed by namespace+gateway\n    privateByNamespaceAndGateway map[types.NamespacedName][]config.Config\n\n    // VS delegation chain: root VS \u2192 delegate VS references\n    delegates map[ConfigKey][]ConfigKey\n}\n```\n\n**Use case**: When a proxy in namespace \"default\" requests \"api.default.svc.cluster.local\", the generator queries `HostnameAndNamespace[\"api.default.svc.cluster.local\"][\"default\"]` and finds the matching VirtualService in milliseconds.\n\n#### Destination Rule Index\n\n```go\ntype destinationRuleIndex struct {\n    // DRs by namespace (organized by service)\n    namespaceLocal map[string]*consolidatedDestRules\n\n    // DRs exported by namespaces to others\n    exportedByNamespace map[string]*consolidatedDestRules\n\n    // DRs in root namespace (apply mesh-wide if no local override)\n    rootNamespaceLocal *consolidatedDestRules\n}\n```\n\n**Consolidation**: Multiple DestinationRules for the same service are merged into a single `ConsolidatedDestRule`.\n\n### Initialization: `InitContext()`\n\n`InitContext()` is called once per push to populate the PushContext:\n\n```go\nfunc (ps *PushContext) InitContext(env *Environment, oldPushContext *PushContext, pushReq *PushRequest) {\n    ps.initializeMutex.Lock()\n    defer ps.initializeMutex.Unlock()\n\n    // Idempotent: return immediately if already initialized\n    if ps.InitDone.Load() {\n        return\n    }\n\n    // Populate mesh config\n    ps.Mesh = env.Mesh()\n    ps.Networks = env.MeshNetworks()\n\n    // Initialize visibility/export defaults\n    ps.initDefaultExportMaps()\n\n    // Either create new context or incrementally update from old\n    if shouldCreateNew(oldPushContext, pushReq) {\n        ps.createNewContext(env)\n    } else {\n        ps.updateContext(env, oldPushContext, pushReq)\n    }\n\n    ps.InitDone.Store(true)\n}\n```\n\n#### Full Context Creation: `createNewContext()`\n\nWhen triggered by full push (e.g., server restart, mesh-wide config change):\n\n```go\nfunc (ps *PushContext) createNewContext(env *Environment) {\n    ps.initServiceRegistry(env, nil)      // Rebuild service list\n    ps.initKubernetesGateways(env)        // K8s Ingress \u2192 Istio Gateway\n    ps.initVirtualServices(env)           // Index all VirtualServices\n    ps.initDestinationRules(env)          // Index all DestinationRules\n    ps.initAuthnPolicies(env)             // Index security policies\n    ps.initAuthorizationPolicies(env)     // Index RBAC policies\n    ps.initTelemetry(env)                 // Index telemetry configs\n    ps.initWasmPlugins(env)               // Index WASM plugins\n    ps.initEnvoyFilters(env, nil, nil)    // Index filter patches\n    ps.initGateways(env)                  // Index Gateways\n    ps.initAmbient(env)                   // Index Ambient mode resources\n    ps.initSidecarScopes(env)             // Index Sidecars (last, depends on others)\n}\n```\n\n**Order matters**: Sidecar scopes are initialized last because they depend on services, VirtualServices, and DestinationRules being available.\n\n#### Incremental Context Update: `updateContext()`\n\nFor partial pushes (e.g., single ServiceEntry added), only changed resources are re-indexed:\n\n```go\nfunc (ps *PushContext) updateContext(env *Environment, oldPushContext *PushContext, pushReq *PushRequest) {\n    // Check which resource types changed\n    for conf := range pushReq.ConfigsUpdated {\n        switch conf.Kind {\n        case kind.ServiceEntry, kind.DNSName:\n            servicesChanged = true\n        case kind.VirtualService:\n            virtualServicesChanged = true\n        case kind.DestinationRule:\n            destinationRulesChanged = true\n        // ... other types\n        }\n    }\n\n    // Reinitialize only changed indexes\n    if servicesChanged {\n        ps.initServiceRegistry(env, pushReq.ConfigsUpdated)\n    } else {\n        ps.ServiceIndex = oldPushContext.ServiceIndex  // Copy unchanged index\n    }\n\n    if virtualServicesChanged {\n        ps.initVirtualServices(env)\n    } else {\n        ps.virtualServiceIndex = oldPushContext.virtualServiceIndex\n    }\n\n    // ... similar for other indexes\n}\n```\n\n### Configuration Translation\n\nThe PushContext doesn't directly generate Envoy resources. Instead, **generators** read from PushContext and produce Envoy protobuf configs:\n\n#### Service \u2192 Cluster (CDS)\n\nWhen generating Clusters, the `cds` generator reads:\n- `PushContext.ServiceIndex` to find the service\n- `PushContext.destinationRuleIndex` to find load balancing policy\n- `PushContext.AuthzPolicies` for authorization context\n\nExample lookup flow:\n```\nRequest: \"Generate CDS for proxy in namespace 'default'\"\n1. Query: ps.ServiceIndex.HostnameAndNamespace[\"api.default.svc.cluster.local\"][\"default\"]\n2. Get Service object with endpoints, ports\n3. Query: ps.destinationRuleIndex for \"api\" in \"default\"\n4. Merge load balancing policy (round-robin, consistent-hash, etc.)\n5. Generate Envoy Cluster protobuf with Endpoints (EDS)\n```\n\n#### VirtualService \u2192 Route (RDS)\n\nWhen generating Routes, the `rds` generator reads:\n- `PushContext.virtualServiceIndex` to find VirtualService\n- Route destinations reference services in PushContext\n- `PushContext.EnvoyFilters` may patch the generated route\n\nExample:\n```\nRequest: \"Generate RDS for proxy in gateway 'istio-system/my-gateway'\"\n1. Query: ps.virtualServiceIndex.publicByGateway[\"istio-system/my-gateway\"]\n2. For each VirtualService:\n   - Extract http/tcp routes\n   - Resolve destination.host to cluster names\n   - Build Envoy RouteConfiguration\n3. Apply EnvoyFilters for patches\n```\n\n#### DestinationRule \u2192 Cluster Subset (CDS)\n\nDestinationRules define subsets of endpoints (e.g., \"v1\", \"v2\" versions):\n\n```\nDestinationRule:\n  spec:\n    subsets:\n    - name: v1\n      labels:\n        version: v1\n    - name: v2\n      labels:\n        version: v2\n```\n\nCDS generator creates separate Cluster for each subset:\n- `api-v1` cluster: endpoints with label `version=v1`\n- `api-v2` cluster: endpoints with label `version=v2`\n\nQueries PushContext:\n- `ps.destinationRuleIndex` for the DestinationRule\n- `ps.ServiceIndex.instancesByPort` for labeled endpoints\n\n### Push Metrics and Error Tracking\n\nPushContext tracks push statistics:\n\n```go\ntype ProxyPushStatus struct {\n    PushStartTime      int64\n    PushDuration       int64  // milliseconds\n    Err                string\n    // ... per-type status\n}\n```\n\nCaptured during push:\n- Successful resource generation\n- Errors (missing services, invalid config, etc.)\n- Per-proxy metrics for debugging\n\n### Relationship to Service Registry\n\nPushContext reads from the service registry via `env.ServiceRegistry()` and `env.GetServices()`:\n- **Services** come from: Kubernetes Services + ServiceEntry resources\n- **Endpoints** come from: Kubernetes Endpoints/EndpointSlice + ServiceEntry\n- **Service Accounts** come from: Kubernetes Pods + manual service account mapping\n\nPush trigger flow:\n```\nService Registry Event (e.g., pod added)\n\u2193\nConfigUpdate() \u2192 pushChannel\n\u2193\ndebounce (100ms - 10s)\n\u2193\nPush() \u2192 initPushContext() \u2192 new PushContext\n\u2193\nPushContext.initServiceRegistry() queries registry\n\u2193\nGenerators use PushContext.ServiceIndex\n\u2193\nxDS resources generated and sent to proxies\n```\n\n---\n\n## 4. Service Registry and Multi-Cluster Support\n\n**Source Files:** `pilot/pkg/serviceregistry/`, `pilot/pkg/serviceregistry/kube/controller/`\n\n### Purpose\n\nThe Service Registry abstracts away the underlying service discovery source. Pilot can aggregate services from:\n- **Kubernetes**: Services, Endpoints, Pods\n- **Service Entries**: Manual service definitions\n- **Multi-cluster**: Services from remote clusters\n- **VMs**: WorkloadEntry resources for VMs outside Kubernetes\n\n### ServiceRegistry Interface\n\n```go\ntype Instance interface {\n    model.Controller                // Event handlers: AppendServiceHandler, Run()\n    model.ServiceDiscovery          // Query methods: GetServices(), InstancesByPort(), etc.\n    Provider() provider.ID          // \"kubernetes\", \"workload\", etc.\n    Cluster() cluster.ID            // Cluster ID for multi-cluster\n}\n```\n\n**Controller** provides event-driven notifications:\n- `AppendServiceHandler(f ServiceHandler)` \u2014 register callback for service changes\n- `Run(stop <-chan struct{})` \u2014 start the controller\n\n**ServiceDiscovery** provides query interface:\n- `GetServices()` \u2014 list all services\n- `GetService(hostname)` \u2014 lookup specific service\n- `InstancesByPort(service, port, labels)` \u2014 find endpoints for a service\n\n### Kubernetes Service Registry Implementation\n\n**Source:** `pilot/pkg/serviceregistry/kube/controller/controller.go`\n\nThe Kubernetes controller watches:\n\n```go\ntype Controller struct {\n    // Watched Kubernetes resources\n    namespaces kclient.Client[*v1.Namespace]    // Watch namespaces\n    services   kclient.Client[*v1.Service]      // Watch k8s Services\n    endpoints *endpointSliceController          // Watch EndpointSlices\n    nodes     kclient.Client[*v1.Node]          // Watch nodes (for gateway addresses)\n    pods      *PodCache                         // Watch Pods (for endpoints, RBAC)\n\n    // Conversion to Istio model\n    servicesMap map[host.Name]*model.Service   // Kubernetes Service \u2192 Istio Service\n    nodeInfoMap map[string]kubernetesNode      // Node info for NodePort IPs\n\n    // Event handlers\n    handlers model.ControllerHandlers            // Registered handlers from PushContext\n}\n```\n\n#### Service Discovery Flow\n\n**1. Service Watch**\n\nWhen a Kubernetes Service is created:\n```\nK8s Event: Service \"api\" in namespace \"default\"\n\u2193\nController.onServiceUpdate()\n\u2193\nConverts: v1.Service \u2192 model.Service\n  - Extract name, namespace, ports, labels, selectors\n  - Create hostname: \"api.default.svc.cluster.local\"\n\u2193\nUpdates: Controller.servicesMap[\"api.default.svc.cluster.local\"]\n\u2193\nFires: ServiceHandler callbacks\n```\n\n**2. Endpoint Discovery**\n\nWhen pods are added/removed, EndpointSlices are updated:\n```\nK8s Event: EndpointSlice updated (new pod ready)\n\u2193\nController.onEndpointUpdate()\n\u2193\nQueries: Pod cache for pod metadata (labels, node, IP)\n\u2193\nResolves: K8s Service selector \u2192 matching pods\n\u2193\nCreates: model.IstioEndpoint entries (IP, port, labels, zone, ...)\n\u2193\nFires: ServiceHandler with updated service\n```\n\n**3. Push Trigger**\n\nService change triggers configuration push:\n```go\nfunc (c *Controller) onServiceUpdate(oldSvc, newSvc *v1.Service, event model.Event) {\n    // Determine what changed (ports, labels, selectors, etc.)\n\n    // Trigger push for affected services\n    c.opts.XDSUpdater.ConfigUpdate(&model.PushRequest{\n        Full: true,  // or false for EDS-only push\n        ConfigsUpdated: map[model.ConfigKey]bool{\n            {Kind: kind.Service, Name: \"api\", Namespace: \"default\"}: true,\n        },\n        Reason: model.NewReasonStats(model.ServiceUpdate),\n    })\n}\n```\n\n#### CDS vs. EDS Push\n\nService changes can trigger different push types:\n\n**Full Push (CDS + LDS + RDS + EDS)**:\n- Service ports changed (requires new Envoy Clusters)\n- Service selectors changed (different endpoints)\n- Service labels changed (affects traffic policies)\n- Service visibility changed (exportTo)\n\n**EDS-Only Push**:\n- Pod scaled up/down (new endpoint IP, same service definition)\n- Pod health changed (endpoint removed from available set)\n- Pod labels changed (subset routing affected)\n\nEDS is much lighter weight than full push:\n- No new Cluster generation\n- No new Route generation\n- Just update endpoint list in existing Cluster\n\n### Multi-Cluster Service Aggregation\n\n**Source:** `pilot/pkg/serviceregistry/kube/controller/multicluster.go`, `pilot/pkg/serviceregistry/aggregate/`\n\nIn multi-cluster deployments, Pilot aggregates services from multiple clusters:\n\n```\nCluster A: K8s Controller\n    \u2514\u2500 Services: api.default, web.default\n       Endpoints: [10.0.1.2:8080, 10.0.1.3:8080]\n\nCluster B: K8s Controller\n    \u2514\u2500 Services: api.default, web.default\n       Endpoints: [10.0.2.2:8080, 10.0.2.3:8080]\n\n       \u2193 (via ServiceExport/ServiceImport or Istio ServiceEntry)\n\nAggregate Controller:\n    \u2514\u2500 api.default: [10.0.1.2, 10.0.1.3, 10.0.2.2, 10.0.2.3]\n       (with cluster labels on each endpoint)\n```\n\n#### Multi-Cluster Modes\n\n**1. Shared Control Plane (Pilot runs in Cluster A)**\n- Pilot watches Kubernetes in Cluster A and reads remote Cluster B via API server or agent\n- Uses `MeshNetworks` config to route traffic across clusters\n- Endpoints are labeled with cluster: \"A\" or \"B\"\n- Gateway pods in Cluster B handle cross-cluster ingress\n\n**2. Shared Network**\n- All pods on same network (no NAT)\n- Proxies directly reach endpoints in remote clusters\n- Endpoints include ClusterName label for debugging\n\n**3. Cross-Network**\n- Each cluster on separate network (NAT)\n- Ingress gateway in remote cluster exposes services\n- Endpoints point to gateway IP:port instead of pod IPs\n\n#### ServiceExport and ServiceImport\n\nKubernetes ServiceExport/ServiceImport resources control multi-cluster visibility:\n\n```yaml\n# Cluster A\napiVersion: net.gke.io/v1\nkind: ServiceExport\nmetadata:\n  name: api\n  namespace: default\n---\n# Cluster B (MCS controller syncs ServiceExport)\napiVersion: net.gke.io/v1\nkind: ServiceImport\nmetadata:\n  name: api\n  namespace: default\nspec:\n  ports:\n  - port: 8080\n  ips:\n  - 10.0.1.1  # Ingress gateway IP in Cluster A\n```\n\n### Workload Entries and Auto-Registration\n\n**Purpose**: Support VMs and non-Kubernetes workloads in the Istio mesh\n\n**Source:** `pilot/pkg/autoregistration/`\n\nWorkloadEntry resources define VMs:\n```yaml\napiVersion: networking.istio.io/v1alpha3\nkind: WorkloadEntry\nmetadata:\n  name: vm-api-1\nspec:\n  address: 192.168.1.10\n  ports:\n    http: 8080\n  labels:\n    app: api\n    version: v1\n```\n\nWorkloadEntry \u2192 Service flow:\n1. Pilot's workload controller watches WorkloadEntry CRD\n2. Creates `model.IstioEndpoint` for the VM\n3. Associates it with a Service (via matching label selectors)\n4. Includes in EDS response to proxies\n\nAuto-registration (for non-Kubernetes proxies):\n- Proxy connects to Pilot with metadata (VM IP, port, labels)\n- Pilot auto-creates WorkloadEntry\n- Subsequently treated like any other endpoint\n\n### Dependency Between Registry and DiscoveryServer\n\nThe push flow connects service changes to xDS updates:\n\n```\nService Registry        PushContext         DiscoveryServer        Proxies\n(K8s, ServiceEntry)\n        \u2502\n        \u251c\u2500 ServiceHandler fired\n        \u2502       \u2502\n        \u2502       \u2514\u2500 ConfigUpdate(pushRequest)\n        \u2502               \u2502\n        \u2502               \u251c\u2500 pushChannel (debounce)\n        \u2502               \u2502\n        \u2502               \u2514\u2500 Push()\n        \u2502                   \u2502\n        \u2502                   \u251c\u2500 initPushContext()\n        \u2502                   \u2502   \u2514\u2500 ServiceRegistry.GetServices()  \u2190 New query\n        \u2502                   \u2502\n        \u2502                   \u2514\u2500 AdsPushAll()\n        \u2502                       \u2514\u2500 pushConnection()\n        \u2502                           \u2514\u2500 pushXds() (CDS, EDS, LDS, RDS)\n        \u2502                               \u2514\u2500 Send to proxy\n        \u2514\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2192\n```\n\n**Key point**: Every service change query the registry fresh (via `InitContext` \u2192 `initServiceRegistry`). PushContext doesn't cache service data across pushes.\n\n---\n\n## 5. Data Flow: Configuration Change to xDS Push\n\n### End-to-End Example: Adding a Pod\n\n```\n1. Kubernetes Event: Pod \"api-v2-xyz\" created\n   - Ready status: Running\n   - IP: 10.0.1.5\n   - Labels: app=api, version=v2\n   - Service selector match: app=api\n\n2. Kube Controller Detects Change\n   - Watch fires for EndpointSlice\n   - Queries Pod cache for pod metadata\n   - Creates IstioEndpoint for 10.0.1.5:8080\n\n3. Push Trigger\n   - ServiceHandler callback from kube controller\n   - Calls ConfigUpdate() with reason: model.ServiceUpdate\n   - Push request goes to pushChannel\n\n4. Debounce\n   - Default: wait 100ms after last change\n   - Max: 10 seconds absolute\n   - If no new changes in 100ms, proceed to push\n\n5. PushContext Initialization\n   - initPushContext() called by Push()\n   - PushContext.initServiceRegistry() queries registry\n   - ServiceIndex rebuilt with new endpoint\n   - Lookup: ps.ServiceIndex[\"api.default\"][8080] \u2192 includes 10.0.1.5\n\n6. xDS Generation\n   - For each connected proxy:\n     - Call pushConnection() with PushContext\n     - CDS generator: checks if proxy needs cluster \"api\" (yes)\n     - EDS generator: loads endpoints from PushContext.ServiceIndex\n     - Generates EDS update with [10.0.1.2, 10.0.1.3, 10.0.1.5]\n     - Send: DiscoveryResponse with new EDS version\n\n7. Proxy Update\n   - Envoy proxy receives EDS response\n   - Adds 10.0.1.5 to connection pool\n   - Starts health checking\n   - Traffic gradually shifts to new endpoint\n```\n\n### Example: Applying VirtualService\n\n```\n1. Istio Event: VirtualService \"api\" created\n   kubectl apply -f virtualservice.yaml\n\n2. Config Watcher Detects Change\n   - ConfigStore watches VirtualService CRD\n   - Fires ConfigUpdate() with reason: model.ConfigChange\n   - ConfigsUpdated: {Kind: VirtualService, Name: \"api\", Namespace: \"default\"}\n\n3. Push Trigger\n   - pushChannel receives PushRequest\n   - Debounce waits for stability\n   - Push() called\n\n4. PushContext Initialization\n   - initPushContext() called\n   - updateContext() detects virtualServicesChanged = true\n   - initVirtualServices() re-indexes all VS\n   - virtualServiceIndex rebuilt with new VS\n\n5. xDS Generation\n   - For each connected proxy:\n     - Determine which proxies care about \"api\" VS\n       - Proxy in gateway namespace: yes\n       - Proxy in other namespace with exportTo: yes\n       - Proxy in private namespace: only if vs.exportTo includes it\n     - RDS generator:\n       - Looks up vs in virtualServiceIndex\n       - Extracts routes from vs.http[]/tcp[]\n       - For each route destination, resolve service to cluster name\n       - Generate Envoy RouteConfiguration\n     - LDS generator:\n       - May need new listener if vs.hosts changed significantly\n     - Send: DiscoveryResponse with new RDS/LDS versions\n\n6. Proxy Update\n   - Envoy receives RDS with new routes\n   - Traffic to \"api\" now respects defined routes\n   - Headers, timeouts, retries applied\n```\n\n---\n\n## 6. Extension Points\n\n### 1. Custom xDS Generators\n\n**How**: Register in `DiscoveryServer.Generators` map\n\n**Why**: Customize resource generation per client type\n\n**Example**: Different configuration for different Envoy versions or proxy types\n\n```go\ngenerators := map[string]model.XdsResourceGenerator{\n    \"envoy-v1\": &CustomGeneratorV1{},\n    \"envoy-v2\": &CustomGeneratorV2{},\n}\ndiscoveryServer.Generators = generators\n```\n\nGenerators implement:\n```go\ntype XdsResourceGenerator interface {\n    Generate(proxy *Proxy, push *PushContext,\n             req *model.PushRequest, w *model.WatchedResource) model.Resources\n}\n```\n\n### 2. Custom ProxyNeedsPush Filter\n\n**How**: Set `DiscoveryServer.ProxyNeedsPush` function\n\n**Why**: Skip pushes for certain proxies (e.g., workload-specific pushes)\n\n```go\ndiscoveryServer.ProxyNeedsPush = func(proxy *Proxy, req *PushRequest) (*PushRequest, bool) {\n    // Return true to send push, false to skip\n    if !proxyInterested(proxy, req) {\n        return nil, false\n    }\n    return req, true\n}\n```\n\n### 3. Custom Service Discovery Backend\n\n**How**: Implement `serviceregistry.Instance` interface and register\n\n**Why**: Add non-Kubernetes sources (Consul, Cloud Run, custom registry)\n\n```go\ntype CustomRegistry struct {}\n\nfunc (cr *CustomRegistry) GetServices() []*model.Service {\n    // Query custom backend\n}\n\nfunc (cr *CustomRegistry) Provider() provider.ID {\n    return \"custom\"\n}\n\n// Register with Pilot\nenv.ServiceRegistry().RegisterRegistry(customRegistry)\n```\n\n### 4. EnvoyFilter Patches\n\n**How**: Apply EnvoyFilter resources\n\n**Why**: Modify generated Envoy config without changing core generators\n\n```yaml\napiVersion: networking.istio.io/v1alpha3\nkind: EnvoyFilter\nmetadata:\n  name: custom-filter\nspec:\n  workloadSelector:\n    labels:\n      app: api\n  configPatches:\n  - applyTo: CLUSTER\n    patch:\n      operation: MERGE\n      value:\n        outlierDetection:\n          consecutive5xx: 5\n```\n\n### 5. WASM Plugin Injection\n\n**How**: Create WasmPlugin resource\n\n**Why**: Inject custom WASM modules into proxy initialization\n\n```yaml\napiVersion: extensions.istio.io/v1alpha1\nkind: WasmPlugin\nmetadata:\n  name: auth-filter\nspec:\n  selector:\n    matchLabels:\n      app: api\n  url: oci://myregistry.com/auth-filter:v1.0\n  phase: AUTHN\n  config: |\n    {\n      \"iss\": \"https://auth.example.com\"\n    }\n```\n\n### 6. Custom Event Handlers\n\n**How**: Implement `model.ServiceHandler` and append to controller\n\n**Why**: React to service changes (custom logging, metrics, external integrations)\n\n```go\ncontroller.AppendServiceHandler(func(old, new *model.Service, event model.Event) {\n    // React to service changes\n    if event == model.EventAdd {\n        log.Infof(\"New service: %s\", new.Hostname)\n    }\n})\n```\n\n### 7. Telemetry and Metrics Plugins\n\n**How**: Implement telemetry generation callbacks\n\n**Why**: Emit custom metrics, logs, or traces\n\n- `PushRequest.Reason` \u2014 tracks push trigger reason\n- `PushContext.ProxyStatus` \u2014 records per-proxy push results\n- Metrics hooks in push flow for custom instrumentation\n\n---\n\n## 7. Error Handling\n\n### Configuration Validation Errors\n\n**Source**: `push_context.go` \u2014 during initialization\n\n**Handling**:\n1. Validator checks config before adding to PushContext\n2. # Istio Pilot Discovery Service Architecture

## Executive Summary

Istio Pilot's Discovery Service is a distributed control plane component that translates Kubernetes-native and Istio-specific network configuration into Envoy sidecar proxy configuration. The service operates as a gRPC server implementing Envoy's xDS (Envoy Discovery Service) protocol, enabling dynamic configuration delivery to a fleet of proxies without restarts.

The system is organized around four primary components:
1. **DiscoveryServer** — the gRPC xDS server managing proxy connections
2. **ADS (Aggregated Discovery Service)** — bidirectional streaming protocol between Pilot and proxies
3. **PushContext** — immutable configuration snapshot used during configuration pushes
4. **ServiceRegistry** — abstraction over service discovery backends (Kubernetes, ServiceEntry, multi-cluster)

---

## 1. DiscoveryServer — The xDS gRPC Server

**Source Files:** `pilot/pkg/xds/discovery.go`, `pilot/pkg/xds/ads.go`

### Purpose

The `DiscoveryServer` is Pilot's core gRPC implementation of Envoy's xDS APIs (CDS, LDS, RDS, EDS, SDS). It serves as the central hub that:
- Maintains active connections to all connected Envoy proxies via `adsClients`
- Coordinates configuration pushes to proxies when Istio resources change
- Implements request rate limiting and push throttling
- Manages cache for xDS resources to avoid recomputation

### Key Responsibilities

#### xDS Protocol Implementation
The DiscoveryServer implements the Envoy xDS v3 protocol with two operational modes:
- **Aggregated Discovery Service (ADS)**: A single bidirectional gRPC stream where all resource types (CDS, LDS, RDS, EDS) flow over one connection
- **Delta XDS**: An incremental update protocol for efficient bandwidth usage
