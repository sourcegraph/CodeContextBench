"""
Rate Limiting Middleware.

This module provides middleware for implementing per-IP request rate limiting
using Django's cache framework.
"""

from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware for rate limiting requests based on client IP address.

    Uses Django's cache framework to track request counts per IP.
    Returns HttpResponseTooManyRequests (429) when the rate limit is exceeded.

    Configuration via settings:
    - RATE_LIMIT_REQUESTS: Number of requests allowed per window (default: 100)
    - RATE_LIMIT_WINDOW: Time window in seconds (default: 3600)
    """

    def process_request(self, request):
        """
        Check if the request exceeds the rate limit for the client's IP.
        """
        # Get rate limit settings with defaults
        limit_requests = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        limit_window = getattr(settings, "RATE_LIMIT_WINDOW", 3600)

        # Get client IP address
        client_ip = request.META.get("REMOTE_ADDR", "unknown")

        # Create cache key for this IP
        cache_key = f"ratelimit:{client_ip}"

        # Get current request count for this IP
        request_count = cache.get(cache_key, 0)

        # Check if limit exceeded
        if request_count >= limit_requests:
            return HttpResponseTooManyRequests()

        # Increment request count
        cache.set(cache_key, request_count + 1, limit_window)

        # Allow the request to proceed
        return None
