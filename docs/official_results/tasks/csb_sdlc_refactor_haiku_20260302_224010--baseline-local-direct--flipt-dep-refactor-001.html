#!/usr/bin/env python3
import re
import sys
import os

def fix_newresource_calls(filepath):
    """
    Fix NewResource() calls to include context.Context as first parameter.
    """
    with open(filepath, 'r') as f:
        content = f.read()
    
    original = content
    
    # Pattern 1: storage.NewResource(
    # Replace with: storage.NewResource(ctx, 
    # We need to be careful and add it at the right place
    
    # Find all functions and their ctx availability
    lines = content.split('\n')
    result = []
    i = 0
    
    while i < len(lines):
        line = lines[i]
        
        # Check if this line contains NewResource call
        if 'NewResource(' in line:
            # We need to insert ctx as first parameter
            # Pattern: NewResource(something -> NewResource(ctx, something
            # Handle both storage.NewResource and just NewResource
            
            # Replace all patterns in this line
            line = re.sub(
                r'NewResource\(',
                'NewResource(ctx, ',
                line
            )
        
        result.append(line)
        i += 1
    
    new_content = '\n'.join(result)
    
    if new_content != original:
        with open(filepath, 'w') as f:
            f.write(new_content)
        return True
    return False

# Find all Go files with NewResource calls
import subprocess

result = subprocess.run(
    ['grep', '-l', 'NewResource(', '-r', '/workspace/internal', '--include=*.go'],
    capture_output=True,
    text=True
)

files = result.stdout.strip().split('\n')
files = [f for f in files if f]

count = 0
for filepath in files:
    if fix_newresource_calls(filepath):
        print(f"Updated: {filepath}")
        count += 1

print(f"\nTotal files updated: {count}")
