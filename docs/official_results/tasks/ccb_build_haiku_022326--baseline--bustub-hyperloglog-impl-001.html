  double m = static_cast<double>(num_buckets_);
  double raw_estimate = 0.0;

  for (size_t i = 0; i < num_buckets_; i++) {
    uint64_t rho = dense_bucket_[i].to_ullong();

    // Check if there's an overflow bucket value
    if (overflow_bucket_.find(i) != overflow_bucket_.end()) {
      // The overflow bucket stores additional rho value beyond the dense bucket
      uint64_t overflow_val = overflow_bucket_[i].to_ullong();
      // Use the max of dense and overflow
      rho = std::max(rho, overflow_val);
    }

    // Clamp rho to reasonable values
    if (rho == 0) {
      rho = 1;
    }

    raw_estimate += std::pow(2.0, -static_cast<double>(rho));
  }

  raw_estimate = CONSTANT * m * m / raw_estimate;

  // Small range correction
  if (raw_estimate <= 2.5 * m) {
    size_t zeros = 0;
    for (size_t i = 0; i < num_buckets_; i++) {
      if (dense_bucket_[i].to_ullong() == 0 && overflow_bucket_.find(i) == overflow_bucket_.end()) {
        zeros++;
      }
    }
    if (zeros != 0) {
      raw_estimate = m * std::log(static_cast<double>(m) / static_cast<double>(zeros));
    }
  }

  // Large range correction
  if (raw_estimate > (1.0 / 30.0) * (1UL << 32)) {
    raw_estimate = -1.0 * (1UL << 32) * std::log(1.0 - raw_estimate / (1UL << 32));
