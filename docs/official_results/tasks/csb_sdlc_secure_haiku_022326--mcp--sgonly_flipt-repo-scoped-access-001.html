5. All changes within `internal/server/evaluation/`

package evaluation

import "sync/atomic"

// Metrics tracks evaluation outcomes in a thread-safe manner.
type Metrics struct {
	totalEvaluations atomic.Uint64
	matchCount       atomic.Uint64
	errorCount       atomic.Uint64
}

// NewMetrics creates a new Metrics tracker.
func NewMetrics() *Metrics {
	return &Metrics{}
}

// RecordEvaluation records a successful evaluation.
func (m *Metrics) RecordEvaluation() {
	m.totalEvaluations.Add(1)
}

// RecordMatch records a successful match evaluation.
func (m *Metrics) RecordMatch() {
	m.matchCount.Add(1)
}

// RecordError records an evaluation error.
func (m *Metrics) RecordError() {
	m.errorCount.Add(1)
}

// GetTotalEvaluations returns the total number of evaluations processed.
func (m *Metrics) GetTotalEvaluations() uint64 {
	return m.totalEvaluations.Load()
}

// GetMatchCount returns the number of successful matches.
func (m *Metrics) GetMatchCount() uint64 {
	return m.matchCount.Load()
}

// GetErrorCount returns the number of evaluation errors.
func (m *Metrics) GetErrorCount() uint64 {
	return m.errorCount.Load()
}
