#pragma once

#include "envoy/http/filter.h"

namespace Envoy {
namespace Extensions {
namespace HttpFilters {
namespace PassThroughFilter {

/**
 * A minimal HTTP filter that passes all data through with Continue status.
 * This filter demonstrates the basic extension pattern for Envoy HTTP filters.
 */
class PassThroughFilter : public Http::StreamDecoderFilter {
public:
  // Http::StreamFilterBase
  void onDestroy() override {}

  // Http::StreamDecoderFilter
  Http::FilterHeadersStatus decodeHeaders(Http::RequestHeaderMap& headers,
                                          bool end_stream) override;
  Http::FilterDataStatus decodeData(Buffer::Instance& data, bool end_stream) override;
  Http::FilterTrailersStatus decodeTrailers(Http::RequestTrailerMap& trailers) override;
  void setDecoderFilterCallbacks(Http::StreamDecoderFilterCallbacks& callbacks) override;

private:
  Http::StreamDecoderFilterCallbacks* decoder_callbacks_{};
};

} // namespace PassThroughFilter
} // namespace HttpFilters
} // namespace Extensions
} // namespace Envoy

#include <memory>
#include <string>

#include "envoy/registry/registry.h"
#include "envoy/server/filter_config.h"

#include "pass_through_filter.h"

namespace Envoy {
namespace Extensions {
namespace HttpFilters {
namespace PassThroughFilter {

/**
 * Config factory for the pass-through filter.
 * Creates filter instances using the NamedHttpFilterConfigFactory pattern.
 */
class PassThroughFilterConfigFactory : public Server::Configuration::NamedHttpFilterConfigFactory {
public:
  Envoy::Http::FilterFactoryCb createFilterFactoryFromProto(
      const Protobuf::Message& proto_config, const std::string& stats_prefix,
      Server::Configuration::FactoryContext& context) override;

  ProtobufTypes::MessagePtr createEmptyConfigProto() override {
    // Return empty proto for this minimal filter that requires no configuration
    return nullptr;
  }

  std::string name() const override { return "envoy.filters.http.pass_through"; }

  bool isTerminalFilter() const override { return false; }
};

Envoy::Http::FilterFactoryCb PassThroughFilterConfigFactory::createFilterFactoryFromProto(
    const Protobuf::Message&, const std::string&, Server::Configuration::FactoryContext&) {
  return [](Http::FilterChainFactoryCallbacks& callbacks) -> void {
    callbacks.addStreamDecoderFilter(std::make_shared<PassThroughFilter>());
  };
}

/**
 * Static registration for the pass-through filter.
 */
REGISTER_FACTORY(PassThroughFilterConfigFactory, Server::Configuration::NamedHttpFilterConfigFactory);

} // namespace PassThroughFilter
} // namespace HttpFilters
} // namespace Extensions
} // namespace Envoy
