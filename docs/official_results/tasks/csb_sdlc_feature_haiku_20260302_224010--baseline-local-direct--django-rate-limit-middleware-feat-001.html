from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware that provides configurable per-IP request rate limiting
    using Django's cache framework.

    Configuration via settings:
    - RATE_LIMIT_REQUESTS: Maximum number of requests allowed per window
                          (default: 100)
    - RATE_LIMIT_WINDOW: Time window in seconds (default: 3600)
    """

    def __init__(self, get_response):
        super().__init__(get_response)
        self.rate_limit_requests = getattr(settings, 'RATE_LIMIT_REQUESTS', 100)
        self.rate_limit_window = getattr(settings, 'RATE_LIMIT_WINDOW', 3600)

    def process_request(self, request):
        """
        Check if the client IP has exceeded the rate limit.
        Returns HttpResponseTooManyRequests (429) if limit exceeded.
        """
        client_ip = request.META.get('REMOTE_ADDR', '')

        # Create a cache key specific to this IP
        cache_key = f'rate_limit:{client_ip}'

        # Get current request count for this IP
        request_count = cache.get(cache_key, 0)

        # Check if limit is exceeded
        if request_count >= self.rate_limit_requests:
            return HttpResponseTooManyRequests()

        # Increment the counter
        # If key doesn't exist, set it with the window timeout
        # If key exists, increment it
        try:
            cache.incr(cache_key)
        except ValueError:
            # Key doesn't exist, set it to 1 with the window timeout
            cache.set(cache_key, 1, self.rate_limit_window)

        return None
