package evaluation

import (
	"sort"
	"sync"
	"time"
)

// DurationTracker records evaluation durations and provides aggregate statistics.
// It is safe for concurrent use.
type DurationTracker struct {
	mu        sync.RWMutex
	durations []time.Duration
}

// Record records a single evaluation duration.
func (dt *DurationTracker) Record(duration time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, duration)
}

// Count returns the total number of recorded evaluations.
func (dt *DurationTracker) Count() int {
	dt.mu.RLock()
	defer dt.mu.RUnlock()
	return len(dt.durations)
}

// Average returns the average duration of all recorded evaluations.
// Returns 0 if no evaluations have been recorded.
func (dt *DurationTracker) Average() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	var total int64
	for _, d := range dt.durations {
		total += int64(d)
	}

	return time.Duration(total / int64(len(dt.durations)))
}

// P99 returns the 99th percentile duration.
// Returns 0 if no evaluations have been recorded.
func (dt *DurationTracker) P99() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	// Create a copy to avoid modifying the original slice
	durations := make([]time.Duration, len(dt.durations))
	copy(durations, dt.durations)

	// Sort the durations
	sort.Slice(durations, func(i, j int) bool {
		return durations[i] < durations[j]
	})

	// Calculate the 99th percentile index
	// For n elements, the 99th percentile is at index ceil(0.99 * n) - 1
	index := (len(durations) * 99 / 100)
	if index >= len(durations) {
		index = len(durations) - 1
	}

	return durations[index]
}
