"""
Rate limiting middleware for Django.

This module provides a middleware that limits the number of requests
from a single IP address within a specified time window.
"""

from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware that enforces rate limiting per IP address.

    Uses Django's cache framework to track request counts.
    Returns 429 (Too Many Requests) when the limit is exceeded.
    """

    def process_request(self, request):
        """
        Check if the request IP has exceeded the rate limit.

        Returns HttpResponseTooManyRequests if limit is exceeded, None otherwise.
        """
        # Get configuration from settings with defaults
        rate_limit_requests = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        rate_limit_window = getattr(settings, "RATE_LIMIT_WINDOW", 3600)

        # Get client IP address
        client_ip = request.META.get("REMOTE_ADDR", "")

        # Create cache key for this IP
        cache_key = f"ratelimit:{client_ip}"

        # Get current request count from cache
        current_count = cache.get(cache_key, 0)

        # Check if limit exceeded
        if current_count >= rate_limit_requests:
            return HttpResponseTooManyRequests()

        # Increment request count
        cache.set(cache_key, current_count + 1, rate_limit_window)

        return None

"""
Rate limiting middleware for Django.

This module provides a middleware that limits the number of requests
from a single IP address within a specified time window.
"""

from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware that enforces rate limiting per IP address.

    Uses Django's cache framework to track request counts.
    Returns 429 (Too Many Requests) when the limit is exceeded.
    """

    def process_request(self, request):
        """
        Check if the request IP has exceeded the rate limit.

        Returns HttpResponseTooManyRequests if limit is exceeded, None otherwise.
        """
        # Get configuration from settings with defaults
        rate_limit_requests = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        rate_limit_window = getattr(settings, "RATE_LIMIT_WINDOW", 3600)

        # Get client IP address
        client_ip = request.META.get("REMOTE_ADDR", "")

        # Create cache key for this IP
        cache_key = f"ratelimit:{client_ip}"

        # Get current request count from cache
        current_count = cache.get(cache_key, 0)

        # Check if limit exceeded
        if current_count >= rate_limit_requests:
            return HttpResponseTooManyRequests()

        # Increment request count
        cache.set(cache_key, current_count + 1, rate_limit_window)

        return None
