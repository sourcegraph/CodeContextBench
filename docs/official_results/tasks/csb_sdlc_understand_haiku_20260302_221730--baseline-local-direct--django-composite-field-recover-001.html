**From:** Platform Team
**Priority:** P2

We need a way to apply validation rules that span multiple form fields simultaneously. Currently, developers must override the form's clean method manually for every cross-field validation.

class CompositeFieldValidator:
    """
    A validator that validates relationships between multiple form fields.

    The validation_function should accept keyword arguments for each field name
    and return True if validation passes, False otherwise.
    """

    def __init__(self, validation_function, field_names, message):
        """
        Initialize the CompositeFieldValidator.

        Args:
            validation_function: A callable that accepts keyword arguments for each
                                field_name and returns True if validation passes.
            field_names: A list/tuple of field names to validate together.
            message: The error message to display when validation fails.
        """
        self.validation_function = validation_function
        self.field_names = field_names
        self.message = message

    def __call__(self, cleaned_data):
        """
        Validate the cleaned form data.

        Args:
            cleaned_data: The form's cleaned_data dictionary.

        Raises:
            ValidationError: If validation fails.
        """
        # Check if all referenced fields exist in cleaned_data
        missing_fields = [
            name for name in self.field_names if name not in cleaned_data
        ]

        if missing_fields:
            # Gracefully handle missing fields
            return

        # Get the values of the specified fields
        field_values = {name: cleaned_data[name] for name in self.field_names}

        # Call the validation function
        if not self.validation_function(**field_values):
            raise ValidationError(self.message)
