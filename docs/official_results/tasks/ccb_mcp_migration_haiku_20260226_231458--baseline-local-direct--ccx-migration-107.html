import os
import re
import json

def check_file_patterns(filepath):
    """Check which legacy patterns are present in a file."""
    patterns = {
        'INITIALIZE_PASS': False,
        'INITIALIZE_PASS_BEGIN_END': False,
        'FunctionPass': False,
        'ModulePass': False,
        'create_factory': False
    }
    
    try:
        with open(filepath, 'r', encoding='utf-8', errors='ignore') as f:
            content = f.read()
            
        # Check for INITIALIZE_PASS patterns
        if 'INITIALIZE_PASS(' in content and not 'INITIALIZE_PASS_BEGIN' in content:
            patterns['INITIALIZE_PASS'] = True
        
        if 'INITIALIZE_PASS_BEGIN' in content or 'INITIALIZE_PASS_END' in content:
            patterns['INITIALIZE_PASS_BEGIN_END'] = True
            
        # Check for class inheritance
        if re.search(r'class\s+\w+\s*:\s*public\s+FunctionPass', content):
            patterns['FunctionPass'] = True
            
        if re.search(r'class\s+\w+\s*:\s*public\s+ModulePass', content):
            patterns['ModulePass'] = True
            
        # Check for create*Pass() functions
        if re.search(r'\bcreate\w+Pass\s*\(', content):
            patterns['create_factory'] = True
            
    except Exception as e:
        print(f"Error reading {filepath}: {e}")
    
    return patterns
