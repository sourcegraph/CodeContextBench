func TestIntegrationScannerDiscovery(t *testing.T) {
	// Setup
	ctx := context.Background()
	tempDir := t.TempDir()

	// Create test audio files in subdirectories
	subDir := filepath.Join(tempDir, "artist", "album")
	if err := os.MkdirAll(subDir, 0o755); err != nil {
		t.Fatalf("Failed to create temp directory: %v", err)
	}

	// Copy fixture files to temp directory
	fixtures := []string{
		"tests/fixtures/test.mp3",
		"tests/fixtures/01 Invisible (RED) Edit Version.mp3",
	}

	for _, fixture := range fixtures {
		src, err := os.Open(fixture)
		if err != nil {
			t.Skipf("Test fixture not found: %s", fixture)
		}
		defer src.Close()

		dst, err := os.Create(filepath.Join(subDir, filepath.Base(fixture)))
		if err != nil {
			t.Fatalf("Failed to create test file: %v", err)
		}
		defer dst.Close()

		if _, err := src.WriteTo(dst); err != nil {
			t.Fatalf("Failed to copy test file: %v", err)
		}
	}

	// Test: Load all audio files from the directory
	files, err := loadAllAudioFiles(subDir)
	if err != nil {
		t.Fatalf("loadAllAudioFiles failed: %v", err)
	}

	// Verify we found the expected number of files
	if len(files) < 2 {
		t.Errorf("Expected at least 2 audio files, got %d", len(files))
	}

	// Verify we can load them by path
	for filePath := range files {
		if !filepath.IsAbs(filePath) {
			t.Errorf("Expected absolute path, got: %s", filePath)
		}
	}
}
